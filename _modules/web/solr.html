def _solr_core_url(config):
    hostname = config['SOLR_HOSTNAME']
    port = config['SOLR_PORT']
    core_name = config['SOLR_CORE_NAME']
    return 'http://{}:{}/solr/{}/'.format(hostname, port, core_name)
